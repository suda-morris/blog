# breakable

import scala.util.control.Breaks.{break, breakable}

// break
breakable {
    for (i <- 1 to 10) {
        if (i == 8) {
            break()
        }
        println("i = " + i)
    }
}

// continue
for (i <- 1 to 10) {
    breakable {
        if (i == 3 || i == 8) {
            break()
        }
        println("i = " + i)
    }
}
