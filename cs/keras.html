import numpy as np

import keras
from keras.models import Sequential  # Keras ä¸­çš„é¡ºåºæ¨¡å‹
from keras.layers import Dense, Activation  # Keras ä¸­çš„å…¨è¿æ¥å±‚å’Œæ¿€æ´»å‡½æ•°
from keras.optimizers import SGD  # Keras ä¸­çš„ SGD ä¼˜åŒ–å™¨
